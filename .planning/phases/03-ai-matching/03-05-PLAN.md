# Plan 03-05: Match Scoring Celery Task

## Objective

Append four async service functions to `matching/service.py`: `find_similar_jobs` (pgvector
cosine similarity query using the HNSW index), `upsert_match` (PostgreSQL ON CONFLICT upsert
into `job_matches`), `list_matches` (ranked match list for a user), and `get_job_match`
(single match lookup). Then append `compute_matches_for_user` to `matching/tasks.py` — the
Celery task that orchestrates profile embedding → cosine search → ATS scoring → bulk upsert.
Finally, add `POST /api/v1/workers/compute-matches` to the workers router.

## Must Haves

- [ ] `find_similar_jobs(db, profile_embedding, limit=50)` returns `list[tuple[Job, float]]` ordered by descending similarity
- [ ] `upsert_match` uses `pg_insert(...).on_conflict_do_update(constraint="uq_job_matches_user_job", ...)` — never select-then-insert
- [ ] `compute_matches_for_user` task populates `job_matches` rows for the given user
- [ ] Task returns `{"user_id": N, "matched": K}` where K is the number of upserted rows
- [ ] `POST /api/v1/workers/compute-matches` with `{"user_id": N}` returns 202 with `{"task_id": "..."}`
- [ ] `uv run pytest tests/ -q --tb=short` passes

Append function `find_similar_jobs`:

```python
async def find_similar_jobs(
    db: AsyncSession,
    profile_embedding: list[float],
    limit: int = 50,
) -> list[tuple[Job, float]]:
    stmt = (
        select(Job, (1 - Job.embedding.cosine_distance(profile_embedding)).label("score"))
        .where(Job.is_active == True)  # noqa: E712
        .where(Job.embedding.is_not(None))
        .order_by(Job.embedding.cosine_distance(profile_embedding))
        .limit(limit)
    )
    result = await db.execute(stmt)
    return [(row.Job, float(row.score)) for row in result]
```

Append function `upsert_match`:

```python
async def upsert_match(
    db: AsyncSession,
    user_id: int,
    job_id: int,
    score: float,
    ats_score: float,
    skill_gaps: list[str],
) -> None:
    now = datetime.now(timezone.utc)
    stmt = pg_insert(JobMatch).values(
        user_id=user_id,
        job_id=job_id,
        score=score,
        ats_score=ats_score,
        skill_gaps=skill_gaps,
        created_at=now,
        updated_at=now,
    )
    stmt = stmt.on_conflict_do_update(
        constraint="uq_job_matches_user_job",
        set_={
            "score": stmt.excluded.score,
            "ats_score": stmt.excluded.ats_score,
            "skill_gaps": stmt.excluded.skill_gaps,
            "updated_at": stmt.excluded.updated_at,
        },
    )
    await db.execute(stmt)
```

Append function `list_matches`:

```python
async def list_matches(
    db: AsyncSession,
    user_id: int,
    limit: int = 50,
    cursor: int | None = None,
) -> list[JobMatch]:
    stmt = (
        select(JobMatch)
        .where(JobMatch.user_id == user_id)
        .order_by(JobMatch.score.desc().nulls_last())
        .limit(limit)
    )
    if cursor is not None:
        stmt = stmt.where(JobMatch.id < cursor)
    result = await db.scalars(stmt)
    return list(result.all())
```

Append function `get_job_match`:

```python
async def get_job_match(
    db: AsyncSession,
    user_id: int,
    job_id: int,
) -> JobMatch | None:
    stmt = select(JobMatch).where(
        JobMatch.user_id == user_id, JobMatch.job_id == job_id
    )
    return await db.scalar(stmt)
```

```python
async def _compute_matches_async(user_id: int, task_id: str) -> dict:
    async with AsyncSessionLocal() as db:
        profile = await db.scalar(select(Profile).where(Profile.user_id == user_id))
        if not profile:
            logger.warning("compute_matches.no_profile", user_id=user_id)
            return {"user_id": user_id, "matched": 0}

        prefs = await db.scalar(
            select(JobPreferences).where(JobPreferences.user_id == user_id)
        )

        profile_text = build_profile_text(profile, prefs)
        if not profile_text.strip():
            logger.warning("compute_matches.empty_profile_text", user_id=user_id)
            return {"user_id": user_id, "matched": 0}

        profile_embedding = await get_embedding(profile_text)

        similar = await find_similar_jobs(db, profile_embedding, limit=50)
        logger.info("compute_matches.found_similar", count=len(similar), user_id=user_id)

        profile_skills: list[str] = profile.skills or []
        for job, score in similar:
            job_text = build_job_text(job)
            ats = compute_ats_score(profile_skills, job_text)
            gaps = compute_skill_gaps(profile_skills, job.requirements)
            await upsert_match(db, user_id, job.id, score, ats, gaps)

        await db.commit()
        logger.info("compute_matches.done", user_id=user_id, matched=len(similar), task_id=task_id)
        return {"user_id": user_id, "matched": len(similar)}
```

Note: `build_job_text` is already imported from `matching.service` in Plan 03-01; ensure the
consolidated import block includes it alongside `find_similar_jobs` and `upsert_match`.

Append the Celery task wrapper:

```python
@celery_app.task(name="jobplatform.matching.tasks.compute_matches_for_user", bind=True)
def compute_matches_for_user(self, user_id: int) -> dict:
    task_id = self.request.id or "local"
    return asyncio.run(_compute_matches_async(user_id, task_id))
```
